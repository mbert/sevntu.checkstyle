26  /** Tests for {@link Murmur3_32HashFunction}. */
27  public class Murmur3Hash32Test extends TestCase {
28    public void testKnownIntegerInputs() {
29      assertHash(593689054, murmur3_32().hashInt(0));
30      assertHash(-189366624, murmur3_32().hashInt(-42));
31      assertHash(-1134849565, murmur3_32().hashInt(42));
32      assertHash(-1718298732, murmur3_32().hashInt(Integer.MIN_VALUE));
33      assertHash(-1653689534, murmur3_32().hashInt(Integer.MAX_VALUE));
34    }

36    public void testKnownLongInputs() {
37      assertHash(1669671676, murmur3_32().hashLong(0L));
38      assertHash(-846261623, murmur3_32().hashLong(-42L));
39      assertHash(1871679806, murmur3_32().hashLong(42L));
40      assertHash(1366273829, murmur3_32().hashLong(Long.MIN_VALUE));

44    public void testKnownStringInputs() {
45      assertHash(0, murmur3_32().hashUnencodedChars(""));
46      assertHash(679745764, murmur3_32().hashUnencodedChars("k"));
47      assertHash(1510782915, murmur3_32().hashUnencodedChars("hell"));
48      assertHash(-675079799, murmur3_32().hashUnencodedChars("hello"));
49      assertHash(1935035788, murmur3_32().hashUnencodedChars("http://www.google.com/"));
50      assertHash(
51          -528633700, murmur3_32().hashUnencodedChars("The quick brown fox jumps over the lazy dog"));
52    }

55    public void testSimpleStringUtf8() {
56      assertEquals(
57          murmur3_32().hashBytes("ABCDefGHI\u0799".getBytes(Charsets.UTF_8)),
58          murmur3_32().hashString("ABCDefGHI\u0799", Charsets.UTF_8));

61    @SuppressWarnings("deprecation")
62    public void testStringInputsUtf8() {
63      Random rng = new Random(0);
64      for (int z = 0; z < 100; z++) {
65        String str;
66        int[] codePoints = new int[rng.nextInt(8)];
67        for (int i = 0; i < codePoints.length; i++) {
68          do {
69            codePoints[i] = rng.nextInt(0x800);
70          } while (!Character.isValidCodePoint(codePoints[i])
71              || (codePoints[i] >= Character.MIN_SURROGATE
72                  && codePoints[i] <= Character.MAX_SURROGATE));
73        }
74        StringBuilder builder = new StringBuilder();
75        for (int i = 0; i < codePoints.length; i++) {
76          builder.appendCodePoint(codePoints[i]);
77        }
78        str = builder.toString();
79        assertEquals(
80            murmur3_32().hashBytes(str.getBytes(Charsets.UTF_8)),
81            murmur3_32().hashString(str, Charsets.UTF_8));
82      }
83    }

85    private static void assertHash(int expected, HashCode actual) {
86      assertEquals(HashCode.fromInt(expected), actual);
87    }

89    public void testParanoidHashBytes() {
90      HashFn hf =
91          new HashFn() {
92            @Override
93            public byte[] hash(byte[] input, int seed) {
94              return murmur3_32(seed).hashBytes(input).asBytes();
95            }
96          };
97      // Murmur3A, MurmurHash3 for x86, 32-bit (MurmurHash3_x86_32)
98      // https://github.com/aappleby/smhasher/blob/master/src/main.cpp
99      HashTestUtils.verifyHashFunction(hf, 32, 0xB0F57EE3);
100   }

102   public void testParanoid() {
103     HashFn hf =
104         new HashFn() {
105           @Override
106           public byte[] hash(byte[] input, int seed) {
107             Hasher hasher = murmur3_32(seed).newHasher();
108             Funnels.byteArrayFunnel().funnel(input, hasher);
109             return hasher.hash().asBytes();
110           }
111         };
112     // Murmur3A, MurmurHash3 for x86, 32-bit (MurmurHash3_x86_32)
113     // https://github.com/aappleby/smhasher/blob/master/src/main.cpp
114     HashTestUtils.verifyHashFunction(hf, 32, 0xB0F57EE3);
115   }
