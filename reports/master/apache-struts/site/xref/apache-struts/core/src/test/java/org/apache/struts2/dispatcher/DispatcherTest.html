178     public void testDispatcherListener() throws Exception {
179 
180     	final DispatcherListenerState state = new DispatcherListenerState();
181 
182     	Dispatcher.addDispatcherListener(new DispatcherListener() {
183 			public void dispatcherDestroyed(Dispatcher du) {
184 				state.isDestroyed = true;
185 			}
186 			public void dispatcherInitialized(Dispatcher du) {
187 				state.isInitialized = true;
188 			}
189     	});
190 
191 
192     	assertFalse(state.isDestroyed);
193     	assertFalse(state.isInitialized);
194 
195         Dispatcher du = initDispatcher(new HashMap<String, String>() );
196 
197     	assertTrue(state.isInitialized);
198 
199     	du.cleanup();
200 
201     	assertTrue(state.isDestroyed);
202     }

241     public void testObjectFactoryDestroy() throws Exception {
242 
243         ConfigurationManager cm = new ConfigurationManager(Container.DEFAULT_NAME);
244         Dispatcher du = new MockDispatcher(new MockServletContext(), new HashMap<String, String>(), cm);
245         Mock mockConfiguration = new Mock(Configuration.class);
246         cm.setConfiguration((Configuration)mockConfiguration.proxy());
247 
248         Mock mockContainer = new Mock(Container.class);
249         String reloadConfigs = container.getInstance(String.class, StrutsConstants.STRUTS_CONFIGURATION_XML_RELOAD);
250         mockContainer.expectAndReturn("getInstance", C.args(C.eq(String.class), C.eq(StrutsConstants.STRUTS_CONFIGURATION_XML_RELOAD)),
251                 reloadConfigs);
252         final InnerDestroyableObjectFactory destroyedObjectFactory = new InnerDestroyableObjectFactory();
253         destroyedObjectFactory.setContainer((Container) mockContainer.proxy());
254         mockContainer.expectAndReturn("getInstance", C.args(C.eq(ObjectFactory.class)), destroyedObjectFactory);
255 
256         mockConfiguration.expectAndReturn("getContainer", mockContainer.proxy());
257         mockConfiguration.expectAndReturn("getContainer", mockContainer.proxy());
258         mockConfiguration.expect("destroy");
259         mockConfiguration.matchAndReturn("getPackageConfigs", new HashMap<String, PackageConfig>());
260 
261         du.init();
262         assertFalse(destroyedObjectFactory.destroyed);
263         du.cleanup();
264         assertTrue(destroyedObjectFactory.destroyed);
265         mockConfiguration.verify();
266         mockContainer.verify();
267     }

316     public void testIsMultipartRequest() throws Exception {
317         MockHttpServletRequest req = new MockHttpServletRequest();
318         HttpServletResponse res = new MockHttpServletResponse();
319 
320         req.setMethod("POST");
321         Dispatcher du = initDispatcher(Collections.<String, String>emptyMap());
322         du.prepare(req, res);
323 
324         req.setContentType("multipart/form-data");
325         assertTrue(du.isMultipartRequest(req));
326 
327         req.setContentType("multipart/form-data; boundary=---------------------------207103069210263");
328         assertTrue(du.isMultipartRequest(req));
329 
330         req.setContentType("multipart/form-data; boundary=---------------------------207103069210263;charset=UTF-8");
331         assertTrue(du.isMultipartRequest(req));
332 
333         req.setContentType("multipart/form-data; boundary=---------------------------207103069210263;charset=ISO-8859-1");
334         assertTrue(du.isMultipartRequest(req));
335 
336         req.setContentType("multipart/form-data; boundary=---------------------------207103069210263;charset=Windows-1250");
337         assertTrue(du.isMultipartRequest(req));
338 
339         req.setContentType("multipart/form-data; boundary=---------------------------207103069210263;charset=US-ASCII");
340         assertTrue(du.isMultipartRequest(req));
341 
342         req.setContentType("multipart/form-data; boundary=---------------------------207103069210263;charset=UTF-16LE");
343         assertTrue(du.isMultipartRequest(req));
344     }

346     public void testServiceActionResumePreviousProxy() throws Exception {
347         Dispatcher du = initDispatcher(Collections.<String, String>emptyMap());
348 
349         MockActionInvocation mai = new MockActionInvocation();
350         ActionContext.getContext().setActionInvocation(mai);
351 
352         MockActionProxy actionProxy = new MockActionProxy();
353         actionProxy.setInvocation(mai);
354         mai.setProxy(actionProxy);
355 
356         mai.setStack(new StubValueStack());
357 
358         HttpServletRequest req = new MockHttpServletRequest();
359         req.setAttribute(ServletActionContext.STRUTS_VALUESTACK_KEY, mai.getStack());
360 
361         assertFalse(actionProxy.isExecutedCalled());
362 
363         du.setDevMode("false");
364         du.setHandleException("false");
365         du.serviceAction(req, null, new ActionMapping());
366 
367         assertTrue("should execute previous proxy", actionProxy.isExecutedCalled());
368     }

370     class InternalConfigurationManager extends ConfigurationManager {
371     	public boolean destroyConfiguration = false;
372 
373         public InternalConfigurationManager(String name) {
374             super(name);
375         }
376 
377         @Override
378     	public synchronized void destroyConfiguration() {
379     		super.destroyConfiguration();
380     		destroyConfiguration = true;
381     	}
382     }
