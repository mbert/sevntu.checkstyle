3   import org.jboss.as.controller.OperationContext;
4   import org.jboss.as.controller.OperationFailedException;
5   import org.jboss.as.controller.OperationStepHandler;
6   import org.jboss.as.controller.PathAddress;
7   import org.jboss.dmr.ModelNode;
8   
9   /**
10   * @author Richard Achmatowicz (c) 2011 Red Hat Inc.
11   */
12  public class TransportLayerRemove implements OperationStepHandler {
13  
14      @Override
15      public void execute(final OperationContext context, final ModelNode operation) throws OperationFailedException {
16          context.removeResource(PathAddress.EMPTY_ADDRESS);
17          reloadRequiredStep(context);
18          context.stepCompleted();
19      }
20  
21      void reloadRequiredStep(final OperationContext context) {
22          if (context.getProcessType().isServer() &&  !context.isBooting()) {
23              context.addStep(new OperationStepHandler() {
24                  @Override
25                  public void execute(final OperationContext context, final ModelNode operation) throws OperationFailedException {
26                      // add some condition here if reload needs to be conditional on context
27                      // e.g. if a service is not installed, don't do a reload
28                      context.reloadRequired();
29                      context.stepCompleted();
30                  }
31              }, OperationContext.Stage.RUNTIME);
32          }
33      }
34  
35  }
