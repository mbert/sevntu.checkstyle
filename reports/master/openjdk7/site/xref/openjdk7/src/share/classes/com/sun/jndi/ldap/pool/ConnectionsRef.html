26  package com.sun.jndi.ldap.pool;
27  
28  /**
29   * Is a reference to Connections that is stored in Pool.
30   * This is an intermediate object that is outside of the circular
31   * reference loop of
32   *  com.sun.jndi.ldap.Connection <-> com.sun.jndi.ldap.LdapClient
33   *    <-> com.sun.jndi.ldap.pool.Connections
34   *
35   * Because Connection is a daemon thread, it will keep LdapClient
36   * alive until LdapClient closes Connection. This will in turn
37   * keep Connections alive. So even when Connections is removed
38   * from (the WeakHashMap of) Pool, it won't be finalized.
39   * ConnectionsRef acts as Connections's finalizer.
40   *
41   * Without connection pooling, com.sun.jndi.ldap.LdapCtx's finalize()
42   * closes LdapClient, which in turn closes Connection.
43   * With connection pooling, ConnectionsRef's finalize() calls
44   * Connections.close(), which in turn will close all idle connections
45   * and mark Connections such that in-use connections will be closed
46   * when they are returned to the pool.
47   */
48  final class ConnectionsRef {
49      final private Connections conns;
50  
51      ConnectionsRef(Connections conns) {
52          this.conns = conns;
53      }
54  
55      Connections getConnections() {
56          return conns;
57      }
58  }
