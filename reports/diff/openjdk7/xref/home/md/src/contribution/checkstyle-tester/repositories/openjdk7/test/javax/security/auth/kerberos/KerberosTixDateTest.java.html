61      public static void main(String[] args) throws Exception {
62          byte[] asn1Bytes = "asn1".getBytes();
63          KerberosPrincipal client = new KerberosPrincipal("client");
64          KerberosPrincipal server = new KerberosPrincipal("server");
65          byte[] keyBytes = "sessionKey".getBytes();
66          long originalTime = 12345678L;
67          Date inDate = new Date(originalTime);
68          boolean[] flags = new boolean[9];
69          flags[8] = true; // renewable
70          KerberosTicket t = new KerberosTicket(asn1Bytes, client, server,
71                  keyBytes, 1 /*keyType*/, flags, inDate /*authTime*/,
72                  inDate /*startTime*/, inDate /*endTime*/,
73                  inDate /*renewTill*/, null /*clientAddresses*/);
74          inDate.setTime(0); // for testing the constructor
75  
76          testDateImmutability(t, originalTime);
77          testS11nCompatibility(t); // S11n: Serialization

80      private static void checkTime(KerberosTicket kt, long timeValue) {
81          if (kt.getAuthTime().getTime() != timeValue) {
82              throw new RuntimeException("authTime check fails!");
83          }
84          if (kt.getStartTime().getTime() != timeValue) {
85              throw new RuntimeException("startTime check fails!");
86          }
87          if (kt.getEndTime().getTime() != timeValue) {
88              throw new RuntimeException("endTime check fails!");
89          }
90          if (kt.getRenewTill().getTime() != timeValue) {
91              throw new RuntimeException("renewTill check fails!");
92          }
93      }
94  
95      private static void testDateImmutability(KerberosTicket t, long origTime)
96          throws Exception {
97          // test the constructor
98          System.out.println("Testing constructor...");
99          checkTime(t, origTime);
100 
101         // test the getAuth/Start/EndTime() & getRenewTill() methods
102         System.out.println("Testing getAuth/Start/EndTime() & getRenewTill()...");
103         t.getAuthTime().setTime(0);
104         t.getStartTime().setTime(0);
105         t.getEndTime().setTime(0);
106         t.getRenewTill().setTime(0);
107         checkTime(t, origTime);
108 
109         System.out.println("DateImmutability Test Passed");
110     }
111 
112     private static void checkEqualsAndHashCode(byte[] bytes, KerberosTicket t)
113         throws IOException, ClassNotFoundException {
114         ByteArrayInputStream bais = new ByteArrayInputStream(bytes);
115         KerberosTicket deserializedTicket = (KerberosTicket)
116                 (new ObjectInputStream(bais).readObject());
117         if (!deserializedTicket.equals(t)) {
118             throw new RuntimeException("equals() check fails!");
119         }
120         if (deserializedTicket.hashCode() != t.hashCode()) {
121             throw new RuntimeException("hashCode() check fails!");
122         }
123     }
124 
125     private static void testS11nCompatibility(KerberosTicket t)
126         throws Exception {
127 
128         System.out.println("Testing against KerberosTicket from JDK6...");
129         byte[] serializedBytes =
130             new BASE64Decoder().decodeBuffer(serializedKerberosTix);
131         checkEqualsAndHashCode(serializedBytes, t);
132 
133         System.out.println("Testing against KerberosTicket from current rel...");
134         ByteArrayOutputStream baos = new ByteArrayOutputStream();
135         new ObjectOutputStream(baos).writeObject(t);
136         checkEqualsAndHashCode(baos.toByteArray(), t);
137 
138         System.out.println("S11nCompatibility Test Passed");
139     }
