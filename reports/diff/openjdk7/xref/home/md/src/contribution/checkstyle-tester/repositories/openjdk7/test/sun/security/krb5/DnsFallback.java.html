23  /*
24   * @test
25   * @bug 6673164
26   * @bug 6552334
27   * @run main/othervm DnsFallback
28   * @summary fix dns_fallback parse error, and use dns by default
29   */
30  
31  import sun.security.krb5.*;
32  import java.io.*;
33  
34  public class DnsFallback {
35      public static void main(String[] args) throws Exception {
36  
37          // for 6673164
38          check("true", "true", true);
39          check("false", "true", false);
40          check("true", "false", true);
41          check("false", "false", false);
42          check("true", null, true);
43          check("false", null, false);
44          check(null, "true", true);
45          check(null, "false", false);
46  
47          // for 6552334
48          check(null, null, true);
49      }
50  
51      static void check(String realm, String fallback, boolean output) throws Exception {
52          FileOutputStream fo = new FileOutputStream("dnsfallback.conf");
53          StringBuffer sb = new StringBuffer();
54          sb.append("[libdefaults]\n");
55          if (realm != null) {
56              sb.append("dns_lookup_realm=" + realm + "\n");
57          }
58          if (fallback != null) {
59              sb.append("dns_fallback=" + fallback + "\n");
60          }
61          fo.write(sb.toString().getBytes());
62          fo.close();
63          System.setProperty("java.security.krb5.conf", "dnsfallback.conf");
64          Config.refresh();
65          System.out.println("Testing " + realm + ", " + fallback + ", " + output);
66          if (Config.getInstance().useDNS_Realm() != output) {
67              throw new Exception("Fail");
68          }
69      }
70  }
