15  package com.google.common.util.concurrent;
16  
17  import static com.google.common.util.concurrent.MoreExecutors.directExecutor;
18  
19  import com.google.common.annotations.GwtIncompatible;
20  import com.google.common.base.Preconditions;
21  import java.util.concurrent.ExecutionException;
22  import java.util.concurrent.Future;
23  import java.util.concurrent.ScheduledExecutorService;
24  import java.util.concurrent.TimeUnit;
25  import java.util.concurrent.TimeoutException;
26  import org.checkerframework.checker.nullness.compatqual.NullableDecl;
27  
28  /**
29   * Implementation of {@code Futures#withTimeout}.
30   *
31   * <p>Future that delegates to another but will finish early (via a {@link TimeoutException} wrapped
32   * in an {@link ExecutionException}) if the specified duration expires. The delegate future is
33   * interrupted and cancelled if it times out.
34   */
35  @GwtIncompatible
36  final class TimeoutFuture<V> extends AbstractFuture.TrustedFuture<V> {
37    static <V> ListenableFuture<V> create(
38        ListenableFuture<V> delegate,
39        long time,
40        TimeUnit unit,
41        ScheduledExecutorService scheduledExecutor) {
42      TimeoutFuture<V> result = new TimeoutFuture<>(delegate);
43      Fire<V> fire = new Fire<>(result);
44      result.timer = scheduledExecutor.schedule(fire, time, unit);
45      delegate.addListener(fire, directExecutor());
46      return result;
47    }
48  
49    /*
50     * Memory visibility of these fields. There are two cases to consider.
51     *
52     * 1. visibility of the writes to these fields to Fire.run:
53     *
54     * The initial write to delegateRef is made definitely visible via the semantics of
55     * addListener/SES.schedule. The later racy write in cancel() is not guaranteed to be observed,
56     * however that is fine since the correctness is based on the atomic state in our base class. The
57     * initial write to timer is never definitely visible to Fire.run since it is assigned after
58     * SES.schedule is called. Therefore Fire.run has to check for null. However, it should be visible
59     * if Fire.run is called by delegate.addListener since addListener is called after the assignment
60     * to timer, and importantly this is the main situation in which we need to be able to see the
61     * write.
62     *
63     * 2. visibility of the writes to an afterDone() call triggered by cancel():
64     *
65     * Since these fields are non-final that means that TimeoutFuture is not being 'safely published',
66     * thus a motivated caller may be able to expose the reference to another thread that would then
67     * call cancel() and be unable to cancel the delegate.
68     * There are a number of ways to solve this, none of which are very pretty, and it is currently
69     * believed to be a purely theoretical problem (since the other actions should supply sufficient
70     * write-barriers).
71     */
72  
73    @NullableDecl private ListenableFuture<V> delegateRef;
74    @NullableDecl private Future<?> timer;
75  
76    private TimeoutFuture(ListenableFuture<V> delegate) {
77      this.delegateRef = Preconditions.checkNotNull(delegate);
78    }
79  
80    /** A runnable that is called when the delegate or the timer completes. */
81    private static final class Fire<V> implements Runnable {
82      @NullableDecl TimeoutFuture<V> timeoutFutureRef;
83  
84      Fire(TimeoutFuture<V> timeoutFuture) {
85        this.timeoutFutureRef = timeoutFuture;
86      }
87  
88      @Override
89      public void run() {
90        // If either of these reads return null then we must be after a successful cancel or another
91        // call to this method.
92        TimeoutFuture<V> timeoutFuture = timeoutFutureRef;
93        if (timeoutFuture == null) {
94          return;
95        }
96        ListenableFuture<V> delegate = timeoutFuture.delegateRef;
97        if (delegate == null) {
98          return;
99        }
100 
101       /*
102        * If we're about to complete the TimeoutFuture, we want to release our reference to it.
103        * Otherwise, we'll pin it (and its result) in memory until the timeout task is GCed. (The
104        * need to clear our reference to the TimeoutFuture is the reason we use a *static* nested
105        * class with a manual reference back to the "containing" class.)
106        *
107        * This has the nice-ish side effect of limiting reentrancy: run() calls
108        * timeoutFuture.setException() calls run(). That reentrancy would already be harmless, since
109        * timeoutFuture can be set (and delegate cancelled) only once. (And "set only once" is
110        * important for other reasons: run() can still be invoked concurrently in different threads,
111        * even with the above null checks.)
112        */
113       timeoutFutureRef = null;
114       if (delegate.isDone()) {
115         timeoutFuture.setFuture(delegate);
116       } else {
117         try {
118           // TODO(lukes): this stack trace is particularly useless (all it does is point at the
119           // scheduledexecutorservice thread), consider eliminating it altogether?
120           timeoutFuture.setException(new TimeoutException("Future timed out: " + delegate));
121         } finally {
122           delegate.cancel(true);
123         }
124       }
125     }
126   }
127 
128   @Override
129   protected String pendingToString() {
130     ListenableFuture<? extends V> localInputFuture = delegateRef;
131     if (localInputFuture != null) {
132       return "inputFuture=[" + localInputFuture + "]";
133     }
134     return null;
135   }
136 
137   @Override
138   protected void afterDone() {
139     maybePropagateCancellationTo(delegateRef);
140 
141     Future<?> localTimer = timer;
142     // Try to cancel the timer as an optimization.
143     // timer may be null if this call to run was by the timer task since there is no happens-before
144     // edge between the assignment to timer and an execution of the timer task.
145     if (localTimer != null) {
146       localTimer.cancel(false);
147     }
148 
149     delegateRef = null;
150     timer = null;
151   }
152 }
