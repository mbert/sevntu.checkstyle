146    /**
147     * A default handler for all commands visited.  This is called for any visit method called, unless a visit command is
148     * appropriately overridden.
149     *
150     * @param ctx     invocation context
151     * @param command command to handle
152     * @return return value
153     * @throws Throwable in the case of a problem
154     */
155    protected Object handleDefault(InvocationContext ctx, VisitableCommand command) throws Throwable {
156       return null;
157    }
158 
159    /**
160     * Helper method to visit a collection of VisitableCommands.
161     *
162     * @param ctx     Invocation context
163     * @param toVisit collection of commands to visit
164     * @throws Throwable in the event of problems
165     */
166    public void visitCollection(InvocationContext ctx, Collection<? extends VisitableCommand> toVisit) throws Throwable {
167       for (VisitableCommand command : toVisit) {
168          command.acceptVisitor(ctx, this);
169       }
170    }
171
