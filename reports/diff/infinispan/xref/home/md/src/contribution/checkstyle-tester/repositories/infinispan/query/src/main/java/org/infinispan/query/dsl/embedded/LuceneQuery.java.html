1   package org.infinispan.query.dsl.embedded;
2   
3   import org.infinispan.query.FetchOptions;
4   import org.infinispan.query.ResultIterator;
5   import org.infinispan.query.dsl.Query;
6   
7   /**
8    * A Query kind that offers iteration and lazy/eager loading options. This is only available in embedded mode.
9    *
10   * @author anistor@redhat.com
11   * @since 6.0
12   */
13  public interface LuceneQuery extends Query, Iterable {
14  
15     /**
16      * Returns the results of a search as a {@link org.infinispan.query.ResultIterator}.
17      * <p/>
18      * Warning: the return type is an extension of {@link java.util.Iterator} which introduces a {@link org.infinispan.query.ResultIterator#close()}
19      * method. This close method needs to be invoked when the iteration is complete to avoid resource leakage.
20      *
21      * @param fetchOptions how to fetch results (see @link FetchOptions)
22      * @return a QueryResultIterator which can be used to iterate through the results that were found.
23      */
24     ResultIterator iterator(FetchOptions fetchOptions);
25  
26     /**
27      * Returns the results of a search as a {@link ResultIterator}. This calls {@link #iterator(FetchOptions fetchOptions)}
28      * with default FetchOptions; this implies eager loading of all results.
29      *
30      * @return a ResultIterator which can be used to iterate through the results that were found.
31      */
32     @Override
33     ResultIterator iterator();
34  }
