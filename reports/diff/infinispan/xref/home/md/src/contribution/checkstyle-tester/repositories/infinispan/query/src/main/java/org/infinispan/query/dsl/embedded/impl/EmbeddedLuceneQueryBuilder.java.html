16   */
17  final class EmbeddedLuceneQueryBuilder extends BaseQueryBuilder<LuceneQuery> {
18  
19     private static final Log log = LogFactory.getLog(EmbeddedLuceneQueryBuilder.class, Log.class);
20  
21     private final QueryEngine queryEngine;
22  
23     EmbeddedLuceneQueryBuilder(QueryFactory queryFactory, QueryEngine queryEngine, String rootType) {
24        super(queryFactory, rootType);
25        this.queryEngine = queryEngine;
26     }
27  
28     @Override
29     public LuceneQuery build() {
30        String jpqlString = accept(new JPAQueryGenerator());
31        if (log.isTraceEnabled()) {
32           log.tracef("JPQL string : %s", jpqlString);
33        }
34        return queryEngine.buildLuceneQuery(queryFactory, jpqlString, startOffset, maxResults, null);
35     }
36  }
