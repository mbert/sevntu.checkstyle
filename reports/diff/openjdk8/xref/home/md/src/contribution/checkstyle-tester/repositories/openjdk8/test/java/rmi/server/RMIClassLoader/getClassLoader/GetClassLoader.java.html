24  /* @test
25   * @bug 4240710
26   * @summary RMIClassLoader.getClassLoader for a codebase should return the
27   * same class loader that is used to load classes from that codebase (using
28   * RMIClassLoader.loadClass).
29   * @author Ann Wollrath
30   *
31   * @library ../../../testlibrary
32   * @build TestLibrary Foo
33   * @run main/othervm/policy=security.policy GetClassLoader
34   */
35  
36  import java.net.*;
37  import java.rmi.*;
38  import java.rmi.server.*;
39  
40  public class GetClassLoader
41  {
42  
43      public static void main(String[] args) {
44  
45          System.err.println("\nTest for rfe 4240710\n");
46  
47          URL codebase1 = null;
48          Class cl = null;
49          ClassLoader loader = null;
50  
51          TestLibrary.suggestSecurityManager(null);
52  
53          try {
54              codebase1 =
55                  TestLibrary.installClassInCodebase("Foo", "codebase1");
56          } catch (MalformedURLException e) {
57              TestLibrary.bomb(e);
58          }
59  
60  
61          /*
62           * Force SecurityException for attempt to obtain a class loader
63           * for a codebase that the test does not have permission to read from.
64           */
65          try {
66              System.err.println(
67                  "getClassLoader for codebase that I can't read");
68              loader = RMIClassLoader.getClassLoader("file:/foo");
69              TestLibrary.bomb(
70                  "Failed: no SecurityException obtaining loader");
71          } catch (MalformedURLException e) {
72              System.err.println(
73                  "Failed: MalformedURLException getting loader");
74              TestLibrary.bomb(e);
75          } catch (SecurityException e) {
76              System.err.println(
77                  "Passed: SecurityException obtaining loader");
78          }
79  
80          /*
81           * Verify that a specific class loaded by name via
82           * RMIClassLoader.loadClass is the same as the class obtained by
83           * loading that class by name through the loader obtained using
84           * the RMIClassLoader.getClassLoader method.
85           */
86          System.err.println("load Foo from codebase1");
87          try {
88              cl = RMIClassLoader.loadClass(codebase1.toString(), "Foo");
89          } catch (Exception e) {
90              System.err.println(
91                  "Failed: exception loading class from codebase1");
92              TestLibrary.bomb(e);
93          }
94  
95          System.err.println(
96              "load Foo using loader obtained using getClassLoader");
97          try {
98              loader = RMIClassLoader.getClassLoader(codebase1.toString());
99          } catch (MalformedURLException e) {
100             System.err.println(
101                 "Failed: MalformedURLException getting codebase1 loader");
102             TestLibrary.bomb(e);
103         }
104 
105         try {
106             if (cl == loader.loadClass("Foo")) {
107                 System.err.println("Passed: Foo classes are equal");
108             } else {
109                 TestLibrary.bomb("Failed: Foo classes are not equal");
110             }
111         } catch (Exception e) {
112             System.err.println(
113                 "Failed: exception loading class from codebase1");
114             TestLibrary.bomb(e);
115         }
116 
117         /*
118          * Force MalformedURLException by passing bogus URL to
119          * getClassLoader.
120          */
121         try {
122             loader = RMIClassLoader.getClassLoader("malformed:///URL");
123             TestLibrary.bomb(
124                 "Failed: getClassLoader should throw MalformedURLException");
125         } catch (MalformedURLException e) {
126             System.err.println(
127                 "Passed: getClassLoader threw MalformedURLException");
128         }
129     }
130 }
